const corrDescriptions = {
  high: '🔥 Near-perfect positive correlation! This is YOUR actual result.',
  medium: '📈 Strong positive correlation. Ads and sales are clearly linked.',
  weak: '📉 Moderate correlation. There is a link but it is not very strong.',
  none: '😐 Very weak or no correlation. Knowing X tells you little about Y.'
};
function updateCorr(val) {
  const v = parseInt(val);
  const labels = ['','Very Loose','Loose','Somewhat Loose','Loose-ish','Moderate','Moderate','Fairly Tight','Tight','Very Tight','Very Tight'];
  document.getElementById('clusterVal').textContent = labels[v] || 'Very Tight';
  const rMap = [0,0.1,0.25,0.4,0.55,0.65,0.75,0.82,0.9,0.97,0.9989];
  const r = rMap[v];
  document.getElementById('corrValue').textContent = r.toFixed(4);
  document.getElementById('rFill').style.width = (r * 100) + '%';
  const r2 = (r*r*100).toFixed(2);
  document.getElementById('r2Value').textContent = r2 + '%';
  drawPie(r*r);
  let desc;
  if(r >= 0.95) desc = corrDescriptions.high;
  else if(r >= 0.7) desc = corrDescriptions.medium;
  else if(r >= 0.4) desc = corrDescriptions.weak;
  else desc = corrDescriptions.none;
  document.getElementById('corrDesc').textContent = desc;
  document.getElementById('r2Desc').innerHTML = `Ad spend explains <strong>${r2}%</strong> of the variation in sales.`;
}

function drawPie(r2) {
  const canvas = document.getElementById('pieCanvas');
  const ctx = canvas.getContext('2d');
  ctx.clearRect(0,0,180,180);
  const cx=90, cy=90, radius=70;
  const angle = r2 * 2 * Math.PI;
  // Background
  ctx.beginPath(); ctx.arc(cx,cy,radius,0,2*Math.PI);
  ctx.fillStyle='rgba(255,255,255,0.06)'; ctx.fill();
  // Explained slice
  ctx.beginPath(); ctx.moveTo(cx,cy);
  ctx.arc(cx,cy,radius,-Math.PI/2,-Math.PI/2+angle);
  ctx.closePath();
  const g = ctx.createRadialGradient(cx,cy,0,cx,cy,radius);
  g.addColorStop(0,'#ffd23f'); g.addColorStop(1,'#38d9a9');
  ctx.fillStyle=g; ctx.fill();
  // Unexplained slice
  ctx.beginPath(); ctx.moveTo(cx,cy);
  ctx.arc(cx,cy,radius,-Math.PI/2+angle,-Math.PI/2+2*Math.PI);
  ctx.closePath(); ctx.fillStyle='rgba(255,255,255,0.06)'; ctx.fill();
  // Labels
  ctx.fillStyle='rgba(255,255,255,0.8)'; ctx.font='bold 13px Poppins'; ctx.textAlign='center';
  ctx.fillText('Explained', cx, cy-10);
  ctx.fillStyle='#ffd23f'; ctx.font='bold 17px Poppins';
  ctx.fillText((r2*100).toFixed(1)+'%', cx, cy+12);
}

function updatePrediction() {
  const x = parseFloat(document.getElementById('predictInput').value) || 0;
  const pred = a_reg + b_reg * x;
  const roi = x > 0 ? (pred / x).toFixed(2) : '—';
  const gain = (pred - x).toFixed(2);
  document.getElementById('predVal').textContent = '₹' + pred.toFixed(2) + 'L';
  document.getElementById('roiVal').textContent = x > 0 ? '₹' + roi : '—';
  document.getElementById('gainVal').textContent = '₹' + gain + 'L';
  document.querySelector('#predMain .pc-desc').textContent = `Using Ŷ = 2.07 + 3.57 × ${x}`;
}

let layers = { dots: true, line: true, labels: true, grid: true };
let hoveredPoint = -1;
let chartCanvas, chartCtx, chartW, chartH;
const pad = {l:60,r:30,t:30,b:60};

function toggleLayer(name, btn) {
  layers[name] = !layers[name];
  btn.classList.toggle('on');
  const colorMap = { dots:'yellow', line:'teal', labels:'blue', grid:'' };
  if(layers[name] && colorMap[name]) btn.classList.add(colorMap[name]);
  else btn.classList.remove('yellow','teal','blue');
  drawScatter();
}

function initChart() {
  chartCanvas = document.getElementById('scatterCanvas');
  chartCtx = chartCanvas.getContext('2d');
  const dpr = window.devicePixelRatio||1;
  chartW = chartCanvas.parentElement.offsetWidth - 64;
  chartH = Math.min(380, window.innerHeight * 0.45);
  chartCanvas.style.width = chartW+'px'; chartCanvas.style.height = chartH+'px';
  chartCanvas.width = chartW*dpr; chartCanvas.height = chartH*dpr;
  chartCtx.scale(dpr,dpr);
  chartCanvas.addEventListener('mousemove', onChartHover);
  chartCanvas.addEventListener('mouseleave', ()=>{ hoveredPoint=-1; drawScatter(); document.getElementById('chartTip').textContent='Hover over a data point to see its values'; });
}

function toX(v) { return pad.l + (v-4)/(13.5-4)*(chartW-pad.l-pad.r); }
function toY(v) { return pad.t + (52-v)/(52-16)*(chartH-pad.t-pad.b); }

function drawScatter() {
  if(!chartCanvas) { initChart(); }
  const ctx = chartCtx;
  ctx.clearRect(0,0,chartW,chartH);

  // Grid
  if(layers.grid) {
    ctx.strokeStyle='rgba(255,255,255,0.05)'; ctx.lineWidth=1;
    [20,30,40,50].forEach(y=>{ctx.beginPath();ctx.moveTo(pad.l,toY(y));ctx.lineTo(chartW-pad.r,toY(y));ctx.stroke();});
    [5,6,7,8,9,10,11,12,13].forEach(x=>{ctx.beginPath();ctx.moveTo(toX(x),pad.t);ctx.lineTo(toX(x),chartH-pad.b);ctx.stroke();});
  }

  // Axes
  ctx.strokeStyle='rgba(255,255,255,0.15)'; ctx.lineWidth=1.5;
  ctx.beginPath(); ctx.moveTo(pad.l,pad.t); ctx.lineTo(pad.l,chartH-pad.b); ctx.lineTo(chartW-pad.r,chartH-pad.b); ctx.stroke();

  // Axis labels
  ctx.fillStyle='rgba(255,255,255,0.4)'; ctx.font='bold 11px Poppins'; ctx.textAlign='right';
  [20,30,40,50].forEach(y=>ctx.fillText('₹'+y+'L',pad.l-10,toY(y)+4));
  ctx.textAlign='center'; ctx.fillStyle='rgba(255,255,255,0.4)';
  dataX.forEach((x,i)=>ctx.fillText(months[i],toX(x),chartH-pad.b+18));
  ctx.fillStyle='rgba(255,255,255,0.3)'; ctx.font='bold 12px Poppins';
  ctx.fillText('Monthly Ad Spend (₹ Lakhs)',(pad.l+chartW-pad.r)/2,chartH-pad.b+40);
  ctx.save(); ctx.translate(16,(pad.t+chartH-pad.b)/2); ctx.rotate(-Math.PI/2);
  ctx.fillText('Sales Revenue (₹ Lakhs)',0,0); ctx.restore();

  // Regression line
  if(layers.line) {
    const grad = ctx.createLinearGradient(toX(4),0,toX(13.5),0);
    grad.addColorStop(0,'rgba(56,217,169,0.3)'); grad.addColorStop(1,'rgba(56,217,169,0.95)');
    ctx.strokeStyle=grad; ctx.lineWidth=2.5; ctx.setLineDash([10,5]);
    ctx.beginPath(); ctx.moveTo(toX(4),toY(a_reg+b_reg*4)); ctx.lineTo(toX(13.5),toY(a_reg+b_reg*13.5)); ctx.stroke();
    ctx.setLineDash([]);
  }

  // Data points
  if(layers.dots) {
    dataX.forEach((x,i)=>{
      const px=toX(x), py=toY(dataY[i]);
      const isHov = hoveredPoint===i;
      // Glow
      if(isHov){
        ctx.beginPath(); ctx.arc(px,py,16,0,Math.PI*2);
        ctx.fillStyle='rgba(255,210,63,0.12)'; ctx.fill();
      }
      // Outer ring
      ctx.beginPath(); ctx.arc(px,py,isHov?10:7,0,Math.PI*2);
      ctx.fillStyle='rgba(255,210,63,0.2)'; ctx.fill();
      // Dot
      ctx.beginPath(); ctx.arc(px,py,isHov?7:5,0,Math.PI*2);
      ctx.fillStyle=isHov?'#fff':'#ffd23f'; ctx.fill();
      ctx.strokeStyle=isHov?'#ffd23f':'rgba(255,107,107,0.6)'; ctx.lineWidth=isHov?2:1.5; ctx.stroke();

      // Labels
      if(layers.labels){
        ctx.fillStyle='rgba(255,255,255,0.5)'; ctx.font='bold 10px Poppins'; ctx.textAlign='center';
        ctx.fillText(months[i],px,py-14);
      }
    });
  }
}

function onChartHover(e) {
  if(!chartCanvas) return;
  const rect = chartCanvas.getBoundingClientRect();
  const mx=(e.clientX-rect.left), my=(e.clientY-rect.top);
  let found=-1;
  dataX.forEach((x,i)=>{
    const px=toX(x), py=toY(dataY[i]);
    if(Math.hypot(mx-px,my-py)<14) found=i;
  });
  if(found!==hoveredPoint){ hoveredPoint=found; drawScatter(); }
  if(found>=0) {
    document.getElementById('chartTip').innerHTML =
      `📅 <strong>${months[found]} 2024</strong> — Ad Spend: <strong style="color:var(--yellow)">₹${dataX[found]}L</strong> → Sales: <strong style="color:var(--teal)">₹${dataY[found]}L</strong>`;
  }
}

let qIdx=0, score=0, answered=false;

function startQuiz() {
  qIdx=0; score=0; answered=false;
  document.getElementById('quizCard').style.display='block';
  document.getElementById('quizScore').classList.remove('show');
  renderQuestion();

function renderQuestion() {
  const q = questions[qIdx];
  document.getElementById('qNum').textContent = `Question ${qIdx+1} of ${questions.length}`;
  document.getElementById('qText').textContent = q.q;
  const opts = document.getElementById('qOptions');
  opts.innerHTML = '';
  q.options.forEach((o,i)=>{
    const btn = document.createElement('button');
    btn.className='quiz-opt'; btn.textContent=o;
    btn.onclick=()=>answerQ(i,btn);
    opts.appendChild(btn);
  });
  document.getElementById('qFeedback').classList.remove('show');
  document.getElementById('qNext').classList.remove('show');
  answered=false;

  const prog = document.getElementById('quizProgress');
  prog.innerHTML='';
  questions.forEach((_,i)=>{
    const dot=document.createElement('div');
    dot.className='qp-dot'+(i<qIdx?' done':i===qIdx?' current':'');
    prog.appendChild(dot);
  });
}

function answerQ(chosen, btn) {
  if(answered) return;
  answered=true;
  const q=questions[qIdx];
  const allOpts=document.querySelectorAll('.quiz-opt');
  allOpts.forEach(b=>b.disabled=true);
  const fb=document.getElementById('qFeedback');
  if(chosen===q.ans){
    score++;
    btn.classList.add('correct');
    fb.className='quiz-feedback show good';
    fb.textContent=q.explain;
  } else {
    btn.classList.add('wrong');
    allOpts[q.ans].classList.add('correct');
    fb.className='quiz-feedback show bad';
    fb.textContent='❌ Not quite. '+q.explain;
  }
  document.getElementById('qNext').classList.add('show');
  document.getElementById('qNext').textContent = qIdx===questions.length-1 ? 'See Results 🏆' : 'Next Question →';
}

    document.getElementById('quizCard').style.display='none';
    const scoreEl=document.getElementById('quizScore');
    scoreEl.classList.add('show');
    const pct=score/questions.length;
    const emojis=['😬','🤔','👍','🎉'];

  } else {
    renderQuestion();
  }
}
